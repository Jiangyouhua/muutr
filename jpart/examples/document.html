    * JNavigation, 站点导航

    let JNavigation = function() {
    PART.apply(this, arguments);
       (function(self, args) {
          let a = Array.prototype.slice.call(args);
          self.SetArgs(a);
       })(this, arguments);
    }
    JNavigation.prototype = new PART()
    JNavigation.prototype.forContent = function() {
       let div = new HTML();
       for(let x = 0; x < this._data.length; x ++) {
          let obj = this._data[x];
          if (!obj.text || !obj.href) {
             console.log("JNavigation Data's text or href is null");
             return;
          }
          let a = new HTML("a", obj.text);
          if(x == parseInt(this._config)){
             a.AddClass("action");
          }
          for(let i in obj){
             if( i == "text"){
             continue;
          }
          a.AddAttr(i, obj[i]);
       }
       if(!this._config) {
          a.AddCss("width", 100 / this._data.length + "%");
          div.AddContent(a);
       }
       this._html = div;
       this._result = this._id;
    }

组件设计

组件设计需要继承PART类，并实现其forContent()方法

为了使用组件具有重用性，您还需要为您的组件添加默认数据及默认设置


一、组件类的基本结构与设计要求

  1. 下面设计一个最简单的组件类
    继承 /**
    * JOther, 组件示例
    * Jiang Youhua 2019.01.10
    */

    let JOther = function() {
       PART.apply(this, arguments);
       (function(self, args) {
          let a = Array.prototype.slice.call(args);
          self.SetArgs(a);
       })(this, arguments);
    }
    JOther.prototype = new PART()
    forContent JOther.prototype.forContent = function() {
       // 判断数据是否有效
       if(!this._data || !this._data.text){
          return;
       }
       // 通过HTML类来实现组件
       let div = new HTML('div', this._data.text);
       for(let x in this._data) {
          if ( x == 'text'){
             continue;
          }
          // 非应用数据默认处理为属性
          div.Attr(x, this._data[x]);
       }
       // 将结果赋予根对象
       this._html = div;
       // 将转入回调函数的数据赋予返回结果
       this._result = this._id;
    }
    默认数据 /**
    * 默认数据
    * text: not null, 文字
    * 其它键值为各项标签属性
    */
    JNavigation.prototype._data = {
       text:'JOther组件示例文字',
       'class':'Other' // 与JS有冲突字段请使用引号
    };
    默认配置 /**
    * 默认配置
    * 该组件没有配置参数
    */
    JNavigation.prototype._data = null;
    引用组件 <script src='./JOther.js' ></script>
    <div data-part='JOther' data-source='' ></div>
    引用组件
    JOther组件示例文字
  2. 组件设计要求，红色项必需严格实现
    PART子类 所有组件必须是PART的子类，否则无法支持
    forContent 所有组件类，必须实现方法同时需要将设计对象赋于this._html
    默认数据 默认数据是让使用者更好的了解该组件需要传入哪种类型的数据
    默认设置 默认设置是让使用者更好的了解设置对组件的影响
    事件完备 组件需要事件支持时需要在组件内完成，尽量减少使用者编辑JS代码
    样式完备 组件需要提供默认的CSS样式以支撑其UI

二、HTML类，输出HTML字符串的类

  1. HTML()，构造函数
    JavaScript let div = new HTML("div", "text", "class='action'");
    Html <div class='action'>text</div>
  2. SetTag()，设置HTML对象的标签
    JavaScript let div = new HTML();
    div.SetTag("p")
    Html <p></p>
  3. SetAttr()，清空HTML对象中所有的属性，并设置新属性
    JavaScript let btn = new HTML("btn", "button");
    div.SetAttr("onclick", "alert(1)")
    Html <button onclick='alert(1)'></button>
  4. AddAttr()，为HTML对象添加新属性
    JavaScript let btn = new HTML("btn", "button");
    btn.AddAttr("type", "button");
    btn.AddAttr("onclick", "alert(1)");
    Html <button onclick='alert(1)' type="button"></button>
  5. SetClass()，清空HTML对象中class属性的所有值，并设置新值
    JavaScript let div = new HTML();
    div.SetClass("action");
    Html <div class="action"></div>
  6. AddClass()，添加HTML对象中class属性的新值
    JavaScript let div = new HTML();
    div.SetClass("action");
    div.AddClass("hot");
    Html <div class="action hot"></div>
  7. SetCss()，空HTML对象中style属性的所有值，并设置新值
    JavaScript let div = new HTML();
    div.SetCss("width", "300px");
    Html <div style="width: 300px"></div>
  8. AddCss()，添加HTML对象中style属性的新值
    JavaScript let div = new HTML();
    div.SetCss("width", "300px");
    div.AddCss("height", "300px");
    Html <div style="width: 300px; height: 300px"></div>
  9. SetContent()，清空HTML对象的内容，并添加新内容
    JavaScript let div = new HTML();
    div.SetContent("text");
    Html <div>text</div>
  10. AddContent()，添加HTML对象的内容
    JavaScript let div = new HTML();
    div.SetContent("text");
    div.AddContent(",txt")
    Html <div>text.txt</div>
  11. Element()，将HTML转为DOM节点，可应于原生的DOM操作中
    JavaScript let div = new HTML();
    div.SetContent("text");
    div.Element()
  12. String()，将HTML转为HTML字符串
    JavaScript let div = new HTML();
    div.SetContent("text");
    div.String()

流程说明

的流程是加载数据经过function处理后输出并调用back function

一、 加载数据的流程是：

  1. 获取data-source属性值，获取值source
  2. 判断该值是否有效？否：退出，是：传入“$”处
  3. “$”处，判断该值的首字母是否为“$”，否：传入url处，是：进入value处
  4. value处，以除首字母的值作JS变量获取值，传入url处
  5. url处，判断传入值是否为url，否：进入节点1，是：进入response处
  6. response处，发起AJAX请求，获取请数据，进入节点1
  7. 获取data-function属性值，获取值function？否：进入节点2，是：传节点1数据返回数据到接点2
  8. 获取data-part属性值，获取值part？否：输出，是：接收有效的data-config属性值和节点1数据后输出
  9. 获取data-back属性值，有:接收part返回值调用该函数
  10. 获取data-affect属性值，有重调该流程
